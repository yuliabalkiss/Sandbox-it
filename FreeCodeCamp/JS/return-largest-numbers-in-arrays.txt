function largestOfFour(arr) {

  let maxNum = [];
 for(let i =0; i <arr.length; i++){
     let a = arr[i][0];
   for (let k =0; k < arr.length; k++){
     if( arr[i][k] > a){
       a = arr[i][k];
      
     }
     
   }
    maxNum.push(a);
 }
 return maxNum;
}
